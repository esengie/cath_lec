sorts =   {Ob, Arr}

fsyms =
          id : Ob -> Arr
          cod, dom : Arr -> Ob
          comp : Arr -> Arr -> Arr
axioms =
          dom(a) = cod(b) |- exists c: comp(a, b) = c -- compE
          comp(a, b) = d ^ comp(d, c) = e ^ comp(b, c) = f ^ comp(a, f) = g |- e = g -- assoc
          comp(id(a), b) = c |- b = c -- idL
          comp(a, id(b)) = c |- a = c -- idR
